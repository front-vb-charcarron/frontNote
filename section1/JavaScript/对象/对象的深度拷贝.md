#对象的深度拷贝
```
	var Chinese = {
		nation : '中国',
		birthPlaces : ['北京','上海','香港']
	};
	
	var Doctor = {
		career : '医生'
	};
	
	function Copy(o,a){
		var c = a || {};
		for(var k in o){
			//判断当前属性类型是否为复杂类型
			if(o[k] instanceof Object){
				//如果是数组，那么下次递归时以数组形式存储数据
				if(o[k].__proto__.constructor == 'Array'){
					var a = [];
					c[k] = Copy(o[k],a);
				}else{
					//否则 当对象处理
					c[k] = Copy(o[k]);
				}
				
			}else{
				c[k] = o[k];
			}
			
		}
		return c;
	}
	var man = Copy(Chinese,null);
	Chinese.birthPlaces[1] = '天津';
			
	console.log(man);
	console.log(Chinese);
```
